TF-IDF

Term Frequency x Inverse Document Frequency. Higher values indicate terms that are uncommon in the corpus but representative in a document.

TF-IDF

count_vect = TfidfVectorizer(
    max_df=0.8,
    min_df=2, 
    stop_words=None)

doc_term_matrix = count_vect.
    fit_transform(r).
    todense()

LDA

LDA = LatentDirichletAllocation(
    n_components=20, 
    random_state=42,
    n_jobs=-1)
LDA.fit(doc_term_matrix)

NMF

nmf = NMF(n_components=5, 
    random_state=1,
    beta_loss='kullback-leibler', 
    solver='mu', 
    max_iter=1000,
    alpha=.1,
    l1_ratio=.5)
nmf.fit(doc_term_matrix)
